    <script>

        // 1. 注册 Service Worker (PWA 必须)

        if ('serviceWorker' in navigator) {

            navigator.serviceWorker.register('./sw.js')

                .then(() => console.log('Service Worker Registered'));

        }

 

        // 2. 模拟获取数据逻辑

        // 注意：由于金十接口有 CORS 保护，前端直接 fetch 会失败。

        // 这里演示两种模式：

        // 模式 A: Demo模式 (生成假数据演示 UI)

        // 模式 B: 真实模式 (你需要填入你自己的后端代理地址)

        

        const MODE = 'REAL'; // 修改为 'REAL' 并填写下面 URL 即可切换

        const PROXY_API_URL = 'https://jin10-proxy.qdshow.workers.dev/';

 

        function createNewsItem(time, content, importance = 'normal') {

            const div = document.createElement('div');

            div.className = 'news-item';

            

            // 简单的关键词高亮逻辑

            let formattedContent = content;

            if (content.includes('利好') || content.includes('上涨')) {

                formattedContent = `<span class="bullish">${content}</span>`;

            } else if (content.includes('利空') || content.includes('下跌')) {

                formattedContent = `<span class="bearish">${content}</span>`;

            } else if (importance === 'high') {

                div.style.borderLeft = '1px solid #ff4d4f';

                formattedContent = `<span class="highlight-important">${content}</span>`;

            }

 

            div.innerHTML = `

                <span class="time">${time}</span>

                <div class="content">${formattedContent}</div>

            `;

            

            const list = document.getElementById('news-list');

            list.insertBefore(div, list.firstChild);

        }

 

        // 模拟数据生成器 (DEMO模式用)

        const demoData = [

            "【行情】现货黄金短线走高5美元，现报2650.45美元/盎司。",

            "美联储主席鲍威尔：通胀数据符合预期，暂不急于降息。",

            "WTI原油日内跌幅扩大至2.00%，现报75.40美元/桶。",

            "【提示】美国10月CPI数据将在十分钟后公布。"

        ];

 

        function fetchNews() {

            if (MODE === 'DEMO') {

                // 随机生成一条快讯

                const now = new Date();

                const timeStr = `${now.getHours().toString().padStart(2,'0')}:${now.getMinutes().toString().padStart(2,'0')}:${now.getSeconds().toString().padStart(2,'0')}`;

                const randomContent = demoData[Math.floor(Math.random() * demoData.length)];

                

                createNewsItem(timeStr, randomContent, Math.random() > 0.8 ? 'high' : 'normal');

            } else {

                // 真实请求逻辑 (需要后端代理)

                fetch(PROXY_API_URL)

                    .then(response => response.json())

                    .then(data => {

                        // 假设返回格式是 { time: '...', content: '...' }

                        createNewsItem(data.time, data.content);

                    })

                    .catch(err => console.error("API Error:", err));

            }

        }

 

        // 初始化

        createNewsItem('系统', '已连接至实时快讯服务器...', 'normal');

        

        // 设置定时器模拟实时推送 (每3秒一条)

        setInterval(fetchNews, 3000);

 

    </script>
